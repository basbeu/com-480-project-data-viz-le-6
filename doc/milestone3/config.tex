\definecolor{skiblue}{RGB}{0,134,255}

\renewcommand{\maketitle}{
\begin{titlepage}
	\includepdf[pagecommand={
		\color{white}
		\vspace*{-1.0cm}
		\begin{spacing}{3.0}
			\begin{center}
				{\fontsize{32}{38}\selectfont
					\thetitle}
			\end{center}
		\end{spacing}
		\vspace*{5cm}
		\begin{spacing}{1.5}
			\begin{center}
				\theauthor
				\vspace*{0.5cm}\\
				\thedate
			\end{center}
		\end{spacing}
	}]
	{img/title.png}
\end{titlepage}
}

%% SECTION FORMAT
\titleformat{\section}
{\normalfont
	\Large
	\bfseries
	\color{white}}
{{\color{skiblue}\thesection{}}}              %% Caption before the name
{2em}                        %% Space between the caption and the name
{\colorbox{skiblue}{#1}}     %% Name of section

\titlespacing{\section}      %% Space around the title
{0pt}{12pt}{12pt}

%% SUBSECTION FORMAT
\titleformat{\subsection}
{\normalfont
	\large
	\bfseries
	\color{skiblue}}
{\thesubsection{}}           %% Caption before the name
{2em}                        %% Space between the caption and the name
{#1}                         %% Name of section

\titlespacing{\subsection}   %% Space around the title
{0pt}{12pt}{12pt}

%% SUBSUBSECTION FORMAT

{0pt}{12pt}{12pt}
